import { describe, it, expect, vi, beforeEach } from "vitest";

// Create a mock chain that tracks calls
function createMockChain(resolveData: any = null, resolveError: any = null) {
  const chain: any = {};
  const calls: { method: string; args: any[] }[] = [];
  const methods = ["from", "select", "insert", "update", "upsert", "eq", "neq", "gt", "gte", "lt", "lte", "order", "limit", "is"];
  for (const m of methods) {
    chain[m] = vi.fn((...args: any[]) => {
      calls.push({ method: m, args });
      return chain;
    });
  }
  chain.single = vi.fn(() => Promise.resolve({ data: resolveData, error: resolveError }));
  chain.maybeSingle = vi.fn(() => Promise.resolve({ data: resolveData, error: resolveError }));
  // Make chain itself thenable for queries without single()
  chain.then = (resolve: any) => resolve({ data: resolveData ? [resolveData] : [], error: resolveError });
  chain._calls = calls;
  return chain;
}

let mockChain: ReturnType<typeof createMockChain>;

vi.mock("@/lib/supabase/server", () => ({
  createClient: vi.fn(() => mockChain),
}));

describe("chat-logger", () => {
  beforeEach(() => {
    vi.resetModules();
    mockChain = createMockChain();
  });

  it("getOrCreateChatSession creates new session when none exists", async () => {
    mockChain = createMockChain(null); // select returns null
    // On insert, return new session
    mockChain.single = vi.fn()
      .mockResolvedValueOnce({ data: null, error: null }) // select
